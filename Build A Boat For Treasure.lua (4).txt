    tweenService, tweenInfo = game:GetService("TweenService"), TweenInfo.new(getgenv().Wait, getgenv().Style)
    tween =
        tweenService:Create(
        game:GetService("Players")["LocalPlayer"].Character:WaitForChild("HumanoidRootPart"),
        tweenInfo,
        {CFrame = CFrame.new(-60.5737877, 58.9498825, 8666.35059)}
    )

        tweenInfo,
        {CFrame = CFrame.new(-55.5486526, -360.063782, 9489.0498)}
    )
    tween:Play()
end
