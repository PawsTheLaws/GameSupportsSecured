local function findClosestPlayer()
    local closestPlayer = nil
    local closestDistance = math.huge
    for _, otherPlayer in pairs(game.Players:GetPlayers()) do
        if otherPlayer ~= player and otherPlayer ~= nil then
            local distance = (player.Character.HumanoidRootPart.Position - otherPlayer.Character.HumanoidRootPart.Position).Magnitude
            if distance < closestDistance then
                closestPlayer = otherPlayer
                closestDistance = distance
            end
        end
    end
    return closestPlayer
end

local function diff(a,b) 
        local localPos = a.Character.HumanoidRootPart.Position
        local plrPos = b.Character.HumanoidRootPart.Position
        
        local dist = (plrPos - localPos).Magnitude
        return dist
end

local function AutoFarm(arg)
    local player = game.Players.LocalPlayer
    local originalCFrame = workspace.machinesFolder[tostring(arg)].interactSeat.CFrame

    local newY = 3500
    local cf = CFrame.new(originalCFrame.Position.x, newY, originalCFrame.Position.z)

    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.CFrame = workspace.machinesFolder[tostring(arg)].interactSeat.CFrame
    end
    workspace.machinesFolder[tostring(arg)].repTime.Value = 0
    local ohString1 = "useMachine"
    local ohInstance2 = workspace.machinesFolder[tostring(arg)].interactSeat
    game:GetService("ReplicatedStorage").rEvents.machineInteractRemote:InvokeServer(ohString1, ohInstance2)
    workspace.machinesFolder[tostring(arg)].interactSeat.CFrame= cf
    task.wait(1)
end
